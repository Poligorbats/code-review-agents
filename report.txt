```python
import typing

def calculate_stats(numbers: typing.List[int]) -> typing.Dict[str, int]:
    """Calculate statistics for a list of numbers."""
    total = sum(numbers)
    avg = total / len(numbers)
    
    result = {
        'total': total,
        'average': avg,
        'count': len(numbers),
        'max': max(numbers),
        'min': min(numbers),
    }
    
    return result


def find_duplicates(l: typing.List[int]) -> typing.List[int]:
    """Find duplicates in a list."""
    duplicates = []
    for i in range(len(l)):
        for j in range(i + 1, len(l)):
            if l[i] == l[j]:
                if l[i] not in duplicates:
                    duplicates.append(l[i])
    
    return duplicates


def read_file(filename: str) -> typing.Optional[str]:
    """Read a file and return its contents."""
    try:
        with open(filename, encoding='utf-8') as f:
            data = f.read()
    
        return data
    except FileNotFoundError:
        return None


def flatten(nested: typing.List[typing.Any], result: typing.List[typing.Any] = []) -> typing.List[typing.Any]:
    """Flatten a nested list."""
    for item in nested:
        if isinstance(item, list):
            flatten(item, result)
        else:
            result.append(item)
    
    return result
```
